\subsection*{1. Vektorlar \( u = (2, 3) \) va \( v = (1, -1) \) ning yig‘indisi qanday?}
\begin{enumerate}[label=\alph*)]
    \item \( (3, 2) \)
    \item \( (1, 4) \)
    \item \( (3, -2) \)
    \item \( (2, -1) \)
